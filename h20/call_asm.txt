	.globl	foo
	.type	foo, @function

	.size	foo, .-foo
	.type	sfoo, @function
sfoo:
	.size	sfoo, .-sfoo
	call	foo

	.type	foo, @function

	.size	foo, .-foo
	.type	sfoo, @function
sfoo:
	.size	sfoo, .-sfoo

	call	efoo@PLT
	call	sfoo
